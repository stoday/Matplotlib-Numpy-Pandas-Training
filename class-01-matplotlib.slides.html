# 畫散佈圖

import random

X_1 = [random.random() for k in range(0, 100)]
y_1 = [random.random() for k in range(0, 100)]

X_2 = [random.random() for k in range(0, 100)]
y_2 = [random.random() for k in range(0, 100)]

plt.figure(figsize=(10, 5))
plt.plot(X_1, y_1, 'b .')
plt.plot(X_2, y_2, 'rx')
plt.legend(['X_1 vs y_1', 'X_2 vs. y_1'])
plt.grid()
plt.show()

ax[0, 0].plot([1, 2, 3], 'r')
